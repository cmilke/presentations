    \underline{Event Selection Criteria:}
    {\footnotesize \begin{itemize}
        \item All jets above 30 GeV
        \item All jets with $|\eta|$ less than 4
        \item All jets pass JVT and fJVT algorithms
        \item Jets truth-matched to photons are excluded\dagger
        \item Must contain at least 2 jets
        \item Leading jet $p_T$ greater than 50 GeV
        \item (Signal Only!) At least two jets must be truth matched to the VBF quarks
    \end{itemize}}
