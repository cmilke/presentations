    \frame{
        \frametitle{Bootstrap Error Formula}

        {\tiny \begin{equation} \begin{split}
        \tilde w &\equiv \textrm{Nominal Weights} \\
        \tilde \alpha &\equiv \textrm{Nominal normalization} \\
        \textrm{IQR}_{w,i} &\equiv \textrm{IQR(25-75) of bootstrap weights for event i} \\
        \textrm{IQR}_{\alpha} &\equiv \textrm{IQR(25-75) of bootstrap normalizations} \\
        H_j &\equiv \textrm{Nominal Histogram Shape} \\
        H'_j &\equiv \textrm{``Varied'' Histogram Shape} \\
        \end{split} \end{equation}}

        {\tiny \begin{equation} \begin{split}
        \sigma_j &\equiv \textrm{Bootstrap Error} \\
        \sigma_j &= | H'_j - H_j | \\
        &= \left| \sum_{i \in j}
        \left[
            \frac{\tilde \alpha \sum \tilde w_i}{\sum 
            \left( 
                \tilde w_i + \frac{1}{2}\textrm{IQR}_{w}
            \right)}
            \left( 
                \tilde w_i + \frac{1}{2}\textrm{IQR}{w}
            \right)
            + \frac{1}{2}\textrm{IQR}_{\alpha}
            \tilde \alpha \tilde w_i
        \right]
        -\tilde \alpha \sum_{i \in j} \tilde w_i \right|
        \end{split} \end{equation}}
    }
